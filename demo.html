üöÄ Usage Examples

Basic Sugar CRM Account Status Enum

import { EnumField, EnumFieldMetadata, required } from '@sugar/ui-react'; const statusMetadata: EnumFieldMetadata = { name: 'account_status', label: 'Account Status', required: true, options: [ { value: 'active', label: 'Active' }, { value: 'inactive', label: 'Inactive' }, { value: 'pending', label: 'Pending' }, { value: 'archived', label: 'Archived' } ], validation: [required('Account status is required')], edit: true, autoCommit: true, help: 'Select the current status of this account' }; function AccountStatusField() { const [status, setStatus] = useState('active'); return ( <EnumField value={status} metadata={statusMetadata} onChange={(value) => setStatus(value)} onCommit={(value) => { // Save to Sugar CRM via API saveToCRM('account_status', value); }} /> ); }

Multi-select Industry Field

const industryMetadata: EnumFieldMetadata = { name: 'target_industries', label: 'Target Industries', multiple: true, required: true, options: [ { value: 'tech', label: 'Technology' }, { value: 'finance', label: 'Financial Services' }, { value: 'healthcare', label: 'Healthcare' }, { value: 'retail', label: 'Retail' } ], display: 'checkbox', validation: [required('At least one industry must be selected')], edit: true, autoCommit: false, help: 'Select all applicable industries' }; function IndustrySelector() { const [industries, setIndustries] = useState(['tech']); return ( <EnumField value={industries} metadata={industryMetadata} onChange={setIndustries} onCommit={(value) => saveToCRM('industries', value)} onValidation={(result) => { if (!result.isValid) { showValidationErrors(result.errors); } }} /> ); }

üîó Sugar CRM Integration

üí° Integration Tip: This component is designed to work seamlessly with Sugar CRM field metadata. You can easily convert existing Sugar field definitions to the component format.
// Example: Converting Sugar CRM field metadata function createEnumMetadata(sugarField) { return { name: sugarField.name, label: sugarField.vname || sugarField.label, required: sugarField.required === true, readonly: !sugarField.editable, options: Object.entries(sugarField.options || {}).map(([value, label]) => ({ value, label: typeof label === 'string' ? label : label.label })), validation: sugarField.required ? [required()] : [], edit: sugarField.mode === 'edit', autoCommit: sugarField.autoCommit !== false, help: sugarField.help || sugarField.comment }; } // Usage in Sugar modules const accountStatusField = createEnumMetadata( App.metadata.getField('Accounts', 'status') ); <EnumField value={this.model.get('status')} metadata={accountStatusField} onChange={(value) => this.model.set('status', value)} onCommit={(value) => this.model.save()} />

üìã API Reference

EnumFieldProps

interface EnumFieldProps { value?: string | number | string[] | number[]; metadata: EnumFieldMetadata; onChange?: (value, metadata) => void; onBlur?: (value, metadata) => void; onFocus?: (metadata) => void; onEdit?: (metadata) => void; onCommit?: (value, metadata) => void; onCancel?: (metadata) => void; onValidation?: (result, metadata) => void; inputProps?: Record<string, any>; }

EnumFieldMetadata

interface EnumFieldMetadata { name: string; // Field name/identifier label: string; // Display label options: EnumOption[]; // Available options required?: boolean; // Whether field is required readonly?: boolean; // Whether field is readonly edit?: boolean; // Whether field is in edit mode multiple?: boolean; // Allow multiple selections display?: 'dropdown' | 'radio' | 'checkbox'; autoCommit?: boolean; // Auto-commit changes placeholder?: string; // Placeholder text help?: string; // Help text validation?: ValidationRule[]; // Validation rules className?: string; // Custom CSS classes }
